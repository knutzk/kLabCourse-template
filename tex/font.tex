\section{Different Font Examples}


\subsection{Subsection with Font Examples (Roman)}

\begin{quotation}
\noindent
\large
\begin{tabular}{l l}
Caps:    & \ABC \\
         & \abc \\
SC:      & \textsc{\abc} \\
Italics: & abcefghijop ~ \textit{abcefghijopy} \\
Ligatures: & Fancy \textit{flesh} fresh \textsc{Blood} final\\
OSF:     & 0123456789 ~~ $0123456789$ \\
\end{tabular}
\end{quotation}
